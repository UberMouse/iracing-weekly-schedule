### Design Direction

> **Aesthetic:** Motorsport telemetry / pit wall dashboard — dark, data-dense, with the precision of racing instrumentation. Think: carbon fiber textures, high-contrast data readouts, the glow of timing screens at night races.

> **Tone:** Industrial-utilitarian meets luxury motorsport. Clean and functional, but with character — not sterile.

**Typography:**
- Display/headings: Use a distinctive condensed or technical font — e.g., `"Barlow Condensed"` or `"Rajdhani"` (Google Fonts). Something that evokes timing boards. NOT Inter, Roboto, or system fonts.
- Body/data: Use a clean sans like `"Barlow"` or `"Outfit"` for readability at small sizes.
- Monospace accents: For track names, week numbers, and data-like elements, use `"JetBrains Mono"` or `"IBM Plex Mono"` for a telemetry feel.

**Color Palette (CSS variables in `index.css`):**
- Background: near-black `#0a0a0f` with subtle blue-tinted depth, not pure gray
- Surface: `#12121a` cards, `#1a1a26` elevated surfaces
- Border: `#2a2a3a` default, `#3a3a4f` hover
- Primary accent: high-visibility amber/yellow `#f5a623` — like pit board signals
- Category colors: each category gets a vivid, saturated accent (not pastel):
  - Oval: electric blue `#3b82f6`
  - Dirt Oval: burnt orange `#f97316`
  - Dirt Road: warm amber `#d97706`
  - Sports Car: racing green `#22c55e`
  - Formula: scarlet red `#ef4444`
- Text: `#e8e8f0` primary, `#8888a0` secondary, `#555570` muted
- Current week highlight: cyan glow `#06b6d4` with `box-shadow` bloom

**Motion (install `motion` package):**
- Page transitions: subtle fade + slight upward slide (150ms)
- Card hover: lift with shadow bloom (`transform: translateY(-2px)`, shadow expands)
- Filter pill toggle: scale bounce on select (spring physics via Motion)
- Modal: backdrop fade-in + modal slides up from below with spring
- Week column highlights: pulse glow animation on current week border
- Stagger card entrance on filter change (50ms delay between cards)

**Spatial & Texture:**
- Subtle noise/grain texture on the main background (CSS pseudo-element with `background-image: url(noise.svg)` or inline SVG data URI)
- Card borders with a subtle gradient that catches light on hover
- Category badges: pill-shaped with the category color at ~15% opacity background, full color text
- Nav bar: frosted glass effect with `backdrop-filter: blur(12px)` and semi-transparent background
- Schedule week columns: subtle inner shadow to create depth, like recessed instrument panels

**Step 3: Implement the store**

Create `src/store/useAppStore.ts`:

```ts
import { create } from "zustand";
import { persist } from "zustand/middleware";
import type { Series, FilterState } from "../types";
import mockData from "../data/mock-schedule.json";

interface AppStore {
  series: Series[];
  favorites: number[];
  toggleFavorite: (seriesId: number) => void;
  weeklyPicks: Record<number, number[]>;
  addWeeklyPick: (week: number, seriesId: number) => void;
  removeWeeklyPick: (week: number, seriesId: number) => void;
  filters: FilterState;
  setFilters: (filters: Partial<FilterState>) => void;
  exportData: () => string;
  importData: (json: string) => void;
}

export const useAppStore = create<AppStore>()(
  persist(
    (set, get) => ({
      series: mockData as Series[],
      favorites: [],
      toggleFavorite: (seriesId) =>
        set((state) => ({
          favorites: state.favorites.includes(seriesId)
            ? state.favorites.filter((id) => id !== seriesId)
            : [...state.favorites, seriesId],
        })),
      weeklyPicks: {},
      addWeeklyPick: (week, seriesId) =>
        set((state) => {
          const current = state.weeklyPicks[week] ?? [];
          if (current.includes(seriesId)) return state;
          return {
            weeklyPicks: { ...state.weeklyPicks, [week]: [...current, seriesId] },
          };
        }),
      removeWeeklyPick: (week, seriesId) =>
        set((state) => ({
          weeklyPicks: {
            ...state.weeklyPicks,
            [week]: (state.weeklyPicks[week] ?? []).filter((id) => id !== seriesId),
          },
        })),
      filters: {
        categories: [],
        licenseClass: null,
        setupType: null,
        searchText: "",
        favoritesOnly: false,
      },
      setFilters: (filters) =>
        set((state) => ({
          filters: { ...state.filters, ...filters },
        })),
      exportData: () => {
        const { favorites, weeklyPicks } = get();
        return JSON.stringify({ favorites, weeklyPicks });
      },
      importData: (json) => {
        const data = JSON.parse(json);
        set({ favorites: data.favorites, weeklyPicks: data.weeklyPicks });
      },
    }),
    {
      name: "iracing-schedule-storage",
      partialize: (state) => ({
        favorites: state.favorites,
        weeklyPicks: state.weeklyPicks,
      }),
    }
  )
);
```

Create `src/components/SeriesCard.tsx`:

```tsx
import { motion } from "motion/react";
import type { Series, Category } from "../types";

interface Props {
  series: Series;
  isFavorite: boolean;
  onToggleFavorite: () => void;
}

const categoryColors: Record<Category, string> = {
  oval: "var(--color-cat-oval)",
  dirt_oval: "var(--color-cat-dirt-oval)",
  dirt_road: "var(--color-cat-dirt-road)",
  sports_car: "var(--color-cat-sports-car)",
  formula: "var(--color-cat-formula)",
};

const categoryLabels: Record<Category, string> = {
  oval: "Oval",
  dirt_oval: "Dirt Oval",
  dirt_road: "Dirt Road",
  sports_car: "Sports Car",
  formula: "Formula",
};

export default function SeriesCard({ series, isFavorite, onToggleFavorite }: Props) {
  const catColor = categoryColors[series.category];

  return (
    <motion.div
      whileHover={{ y: -2 }}
      transition={{ type: "spring", stiffness: 400, damping: 25 }}
      className="bg-[var(--color-surface)] border border-[var(--color-border)] rounded-lg p-4 flex flex-col gap-3 hover:border-[var(--color-border-hover)] hover:shadow-lg hover:shadow-black/20 transition-[border-color,box-shadow] duration-200"
      style={{ borderLeftWidth: "2px", borderLeftColor: catColor }}
    >
      <div className="flex items-start justify-between gap-2">
        <h3 className="font-display font-semibold text-sm leading-tight uppercase tracking-wide">
          {series.seriesName}
        </h3>
        <motion.button
          onClick={onToggleFavorite}
          aria-label="Toggle favorite"
          data-favorited={isFavorite ? "true" : "false"}
          whileTap={{ scale: 1.3 }}
          transition={{ type: "spring", stiffness: 500, damping: 15 }}
          className="shrink-0 text-lg leading-none"
          style={{ color: isFavorite ? "var(--color-accent)" : "var(--color-text-muted)" }}
        >
```

```tsx
import { useMemo } from "react";
import { useAppStore } from "../../store/useAppStore";
import SeriesCard from "../SeriesCard";
import FilterBar from "./FilterBar";

export default function SeriesBrowser() {
  const { series, filters, favorites, toggleFavorite } = useAppStore();

  const filtered = useMemo(() => {
    return series.filter((s) => {
      if (filters.categories.length > 0 && !filters.categories.includes(s.category))
        return false;
      if (filters.licenseClass && s.licenseClass !== filters.licenseClass) return false;
      if (filters.setupType && s.setupType !== filters.setupType) return false;
      if (
        filters.searchText &&
        !s.seriesName.toLowerCase().includes(filters.searchText.toLowerCase())
      )
        return false;
      if (filters.favoritesOnly && !favorites.includes(s.seriesId)) return false;
      return true;
    });
  }, [series, filters, favorites]);

  return (
    <div>
      <FilterBar />
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-4">
        {filtered.map((s) => (
          <SeriesCard
            key={s.seriesId}
            series={s}
            isFavorite={favorites.includes(s.seriesId)}
            onToggleFavorite={() => toggleFavorite(s.seriesId)}
          />
        ))}
      </div>
      {filtered.length === 0 && (
        <p className="text-center text-gray-500 py-12">No series match your filters.</p>
      )}
    </div>
  );
}
```

Create `src/components/ExportImport.tsx`:

```tsx
import { useRef } from "react";
import { useAppStore } from "../store/useAppStore";

export default function ExportImport() {
  const { exportData, importData } = useAppStore();
  const fileInputRef = useRef<HTMLInputElement>(null);

  const handleExport = () => {
    const json = exportData();
    const blob = new Blob([json], { type: "application/json" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = "iracing-schedule.json";
    a.click();
    URL.revokeObjectURL(url);
  };

  const handleImport = () => {
    fileInputRef.current?.click();
  };

  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = (ev) => {
      try {
        importData(ev.target?.result as string);
      } catch {
        alert("Invalid schedule file");
      }
    };
    reader.readAsText(file);
    e.target.value = "";
  };

  return (
    <div className="flex gap-2">
      <button
        onClick={handleExport}
        className="text-xs px-3 py-1.5 rounded-md border border-gray-700 text-gray-400 hover:text-gray-200 hover:border-gray-500 transition-colors"
      >
        Export
      </button>
      <button
        onClick={handleImport}
        className="text-xs px-3 py-1.5 rounded-md border border-gray-700 text-gray-400 hover:text-gray-200 hover:border-gray-500 transition-colors"
      >
        Import
      </button>
      <input ref={fileInputRef} type="file" accept=".json" onChange={handleFileChange} className="hidden" />
    </div>
  );
}
```
